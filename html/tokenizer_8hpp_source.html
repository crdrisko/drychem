90 constexpr value_type nextToken()
91 {
92 value_type str {};
93
94 if (str_current != str_end)
95 {
96 while (str_current != str_end && isDelimiter(*str_current))
97 {
98 if (keepDelimiters && foundSubstr(*str_current, keepDelimiters.value()))
99 return value_type {*str_current++};
100
101 ++str_current;
102 }
103
104 while (str_current != str_end && !isDelimiter(*str_current))
105 str += value_type {*str_current++};
106 }
